```ts
export function shouldLoadDebugPanel(search: string, isDev: boolean): boolean;
// 仅当 isDev && new URLSearchParams(search).get("debug") === "1" → true。只认 "1"。
```

```ts
if (shouldLoadDebugPanel(window.location.search, import.meta.env.DEV)) {
  void import("./debug/panel").then((m) => m.mountDebugPanel({ config, seed }));

```

### 3.4 `src/debug/replay.ts`
按尺子头注：`WaveReplayRow`/`ReplayResult`/`replayBattle(opts)`。实现要点：
- 内部 `generateMap({config, seed})` → `buildDefaultLoadout`(未显式传 loadout 时) → `new CombatSimulation({..., rng: createRng(seed)})` → `new BattleController({config, bus, simulation})`。
- 统计**只来自 EventBus 订阅**：`monster.spawned` 记 entityId→waveIndex；kills/leaks 按怪出生波；`rune.fired` 按 `targetEntityId` 的出生波归属；`battle.settled` 收官。
- 驱动：`battle.start()` 后按 `1000/simulationFps` 步进 `battle.tick(...)` 直至 `phase === "settle"`(加一个非常量步数上限防死循环,上限用 config 派生,如 `simulationFps × 60 × 30`——已在尺子测试中示范)。
- rows 长度 = `config.balance.battle.wavesPerLevel`,waveIndex 1 起。

## 4 · 关键设计(为何这样定)
1. **面板=数值核的第二个消费方**，这是 T6 数值核重构的验收闭环：CLI 与浏览器读同一 `balance-core.mjs`，谁复制公式谁制造漂移(债#2 的教训)。
2. **回放用真实 CombatSimulation 而非解析模型**，因为债#6 正是"解析容量曲线 vs 真实战斗"的落差——面板把两者并排,才能校准 `SIM_SCRIPT.capacity` 的剧本假设(qualityMul/相生/AOE 折算)。
3. **统计按出生波归属**(而非发生时刻的时间轴波)，保证归属规则确定、可独立复算——这是尺子"真值自证"用例的比对基础。
4. **gate 纯函数抽出**是为了让"生产零加载"可被单测钉死，而不是靠人肉瞪 main.ts。
